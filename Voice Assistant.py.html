48   def authenticate_googleCalendar():
49       """Shows basic usage of the Google Calendar API. 
50       Prints the start and name of the next 10 events on the user's calendar. 
51       """
52       creds = None
53       # The file token.json stores the user's access and refresh tokens, and is
54       # created automatically when the authorization flow completes for the first
55       # time.
56       if os.path.exists('token.json'):
57           creds = Credentials.from_authorized_user_file('token.json', SCOPES)
58       # If there are no (valid) credentials available, let the user log in.
59       if not creds or not creds.valid:
60           if creds and creds.expired and creds.refresh_token:
61               creds.refresh(Request())
62           else:
63               flow = InstalledAppFlow.from_client_secrets_file(
64                   'credentials.json', SCOPES)
65               creds = flow.run_local_server(port=0)
66           # Save the credentials for the next run
67           with open('token.json', 'w') as token:
68               token.write(creds.to_json())
69   
70       service = build('calendar', 'v3', credentials=creds)
71   
72       return service

74   def get_events(day, service):
75       # Call the Calendar API
76       date = datetime.datetime.combine(day, datetime.datetime.min.time())
77       end = datetime.datetime.combine(day, datetime.datetime.max.time())
78       utc = pytz.UTC
79       date = date.astimezone(utc)
80       end = end.astimezone(utc)
81       events_result = service.events().list(calendarId = 'primary', timeMin = date.isoformat(),
82                                           timeMax = end.isoformat(), singleEvents = True,
83                                           orderBy = 'startTime').execute()
84       events = events_result.get('items', [])
85   
86       if not events:
87           speak('No upcoming events found.')
88       else:
89           speak(f"You have {len(events)} events on this day.")
90   
91       for event in events:
92           start = event['start'].get('dateTime', event['start'].get('date'))
93           print(start, event['summary'])
94   
95           start_time = str(start.split("T")[1].split("-")[0])  # get the hour the event starts
96           if int(start_time.split(":")[0]) < 12:  # if the event is in the morning
97               start_time = start_time + "am"
98           else:
99               # converting 24-hour time format to 12 hour + |we need to allow the minute aspect to be functional instead of only hour|
100              start_time = str(int(start_time.split(":")[0]) - 12)
101              start_time = start_time + "pm"
102  
103          speak(event["summary"] + " at " + start_time)

106  def get_date(text):
107      text = text.lower()
108      # The following code makes it so that the user can't just request a date that already passed
109      # because it wouldn't make sense to make a request to get a past event.
110      # Instead, it allows the current date the user is making the request as well as the days following it.
111      today = datetime.date.today()
112  
113      if text.count("today") > 0:
114          return today
115  
116      day = -1
117      day_of_week = -1
118      month = -1
119      year = today.year
120  
121      # If the user does not make a request to get their tasks for the current day,
122      # we iterate through this for loop to find out which date they are talking about.
123      for word in text.split():
124          if word in MONTHS:
125              month = MONTHS.index(word) + 1
126          elif word in DAYS:
127              day_of_week = DAYS.index(word)
128  
129          #If the word is simply a digit, make the voice assistant interpret the digit as
130          #date with one of the following extensions: "rd", "th", "st", "nd"
131          elif word.isdigit():
132              day = int(word)
133          else:
134              for ext in DAY_EXTENSIONS:
135                  found = word.find(ext)
136                  # SlICING BY INDEX
137                  if found > 0:
138                      try:
139                          day = int(word[:found])
140                      except:
141                          pass
142  
143      if month < today.month and month != -1:
144          # if the month mentioned is before the current month, set the year to the next
145          year += 1
146  
147      if month == -1 and day != -1:  # if we didn't find a month, but we have a day
148          if day < today.day:
149              month = today.month + 1
150          else:
151              month = today.month
152  
153      # if we only found a date of the week
154      if month == -1 and day == -1 and day_of_week != -1:
155          current_day_of_week = today.weekday()
156          # Find the difference between the two dates
157          dif = day_of_week - current_day_of_week
158          # So that we can start fresh every new week/every Monday
159          if dif < 0:
160              dif += 7
161              if text.count("next") >= 1:
162                  dif += 7
163  
164          return today + datetime.timedelta(dif)
165      # if the user doesn't provide a date in a correct format, they will get something like -1/-1/-1
166      # in order to resolve that we do:
167      if day != -1:
168          return datetime.date(month=month, day=day, year=year)

170  def note(file_name, text):
171      full_path = os.path.abspath(file_name)
172  
173      # Write text to the file
174      with open(full_path, 'w') as file:
175          file.write(text)
176  
177      # Open the file based on the platform
178      if os.path.exists(full_path):
179          if platform.system() == "Darwin":  # macOS
180              subprocess.Popen(["open", full_path])
181          else:
182              print("Unsupported operating system")
183      else:
184          print(f"File not found: {full_path}")
